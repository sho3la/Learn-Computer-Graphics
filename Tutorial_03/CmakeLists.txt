cmake_minimum_required(VERSION 3.9)

project("Tutorial_03")

set(HEADER_FILES
	Window.h
	Points.h
	ShaderProgram.h
	Camera.h
)

set(SOURCE_FILES
	main.cpp
	
	Window.cpp
	Points.cpp
	ShaderProgram.cpp
	Camera.cpp
)

add_executable(tutorial_03
	${HEADER_FILES}
	${SOURCE_FILES}
)

target_link_libraries(tutorial_03
	PUBLIC
		opengl32
		glew32
		glfw3
)

if("${CMAKE_SIZEOF_VOID_P}" EQUAL "8")

	add_custom_command(
    TARGET tutorial_03 POST_BUILD
	
	COMMAND ${CMAKE_COMMAND} -E copy 
	${CMAKE_SOURCE_DIR}/external/glew/bin/x64/glew32.dll
	$<TARGET_FILE_DIR:tutorial_03>
)
else("${CMAKE_SIZEOF_VOID_P}" EQUAL "4")

add_custom_command(
    TARGET tutorial_03 POST_BUILD
	
	COMMAND ${CMAKE_COMMAND} -E copy 
	${CMAKE_SOURCE_DIR}/external/glew/bin/Win32/glew32.dll
	$<TARGET_FILE_DIR:tutorial_03>
)

ENDIF()

if(WIN32)
	target_compile_definitions(tutorial_03
		PRIVATE
			OS_WINDOWS=1
	)
	target_compile_options(tutorial_03
		PRIVATE
			/WX
	)
endif(WIN32)


target_compile_features(tutorial_03 PUBLIC cxx_std_17)
set_target_properties(tutorial_03 PROPERTIES
	CXX_EXTENSIONS OFF
)

target_include_directories(tutorial_03
	PUBLIC
	${CMAKE_SOURCE_DIR}/external/glew/include
	${CMAKE_SOURCE_DIR}/external/glfw/include
	${CMAKE_SOURCE_DIR}/external/glm
)

if("${CMAKE_SIZEOF_VOID_P}" EQUAL "8")

	target_link_directories (tutorial_03
	PUBLIC
	${CMAKE_SOURCE_DIR}/external/glew/lib/x64
    ${CMAKE_SOURCE_DIR}/external/glfw/lib/$<CONFIG>/x64lib/
)	
else("${CMAKE_SIZEOF_VOID_P}" EQUAL "8")

	target_link_directories (tutorial_03
	PUBLIC
	${CMAKE_SOURCE_DIR}/external/glew/lib/Win32
	${CMAKE_SOURCE_DIR}/external/glfw/lib/$<CONFIG>/win32lib/
)
ENDIF()
